  }

import requests
from datetime import datetime

class GroupBrandingManager:
    def __init__(self, auth_token):
        self.headers = {'Authorization': f'Bearer {auth_token}'}
        self.base_url = 'https://api.monitoragrupo.com/v1'
    
    def update_seasonal_branding(self, campaign_id, season):
        """Atualiza branding dos grupos para campanhas sazonais"""
        # Obter grupos da campanha
        response = requests.get(
            f'{self.base_url}/campaigns/{campaign_id}/groups',
            headers=self.headers
        )
        
        groups = response.json()['groups']
        
        # Mapear imagens por esta√ß√£o
        seasonal_images = {
            'black_friday': 'https://cdn.example.com/bf-2026.jpg',
            'natal': 'https://cdn.example.com/natal-2026.jpg',
            'ano_novo': 'https://cdn.example.com/ano-novo-2026.jpg',
            'pascoa': 'https://cdn.example.com/pascoa-2026.jpg'
        }
        
        image_url = seasonal_images.get(season)
        if not image_url:
            raise ValueError(f"Esta√ß√£o '{season}' n√£o configurada")
        
        # Atualizar todos os grupos
        updated = []
        for group in groups:
            response = requests.put(
                f'{self.base_url}/groups/{group["id"]}/image',
                headers=self.headers,
                json={'image_url': image_url}
            )
            
            if response.status_code == 200:
                updated.append(group['name'])
        
        return {
            'season': season,
            'groups_updated': len(updated),
            'timestamp': datetime.now().isoformat()
        }
    
    def apply_ab_test_images(self, campaign_id, image_a, image_b):
        """Aplica teste A/B de imagens nos grupos"""
        response = requests.get(
            f'{self.base_url}/campaigns/{campaign_id}/groups',
            headers=self.headers
        )
        
        groups = response.json()['groups']
        
        # Dividir grupos em dois conjuntos
        half = len(groups) // 2
        group_a = groups[:half]
        group_b = groups[half:]
        
        # Aplicar imagem A
        for group in group_a:
            requests.put(
                f'{self.base_url}/groups/{group["id"]}/image',
                headers=self.headers,
                json={'image_url': image_a, 'ab_test': 'variant_a'}
            )
        
        # Aplicar imagem B
        for group in group_b:
            requests.put(
                f'{self.base_url}/groups/{group["id"]}/image',
                headers=self.headers,
                json={'image_url': image_b, 'ab_test': 'variant_b'}
            )
        
        return {
            'variant_a_groups': len(group_a),
            'variant_b_groups': len(group_b),
            'test_started': datetime.now().isoformat()
        }

# Uso
