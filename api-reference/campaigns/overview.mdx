  const response = await fetch(
    `https://api.monitoragrupo.com/v1/campaigns/${campaignId}/groups`,
    { headers: { 'Authorization': 'Bearer YOUR_TOKEN' } }
  );
  
  const { groups, stats } = await response.json();

    totalCapacity,
    totalMembers,
